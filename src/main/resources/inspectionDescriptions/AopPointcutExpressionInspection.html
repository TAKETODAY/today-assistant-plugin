Reports AspectJ pointcut designators which are not supported by Infra.

Example:


<beans >
  <aop:config >
      <aop:pointcut expression="@this(Anno)  <!-- '@this' pointcut designator isn't supported by Infra -->
                  and adviceexecution()"/> <!--  'adviceexecution' pointcut designator isn't supported by Infra -->
      <aop:pointcut expression="!@annotation(Anno)"/>
    </aop:config>
</beans>
